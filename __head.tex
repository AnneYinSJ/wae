\newcommand{\plotpp}[2]{
\pgfplotsset{width=3.75cm,height=2cm,compat=1.3}
\begin{tikzpicture}
\pgfplotsset{
    scale only axis,
}
\begin{axis}[
	align = center,
	title = {#1},
	smooth,
	axis y line*=left,
	ymin=.4, ymax=1,
	xlabel=Processed chunks,
	%yticklabels = {,\oldstylenums{0}\%,\oldstylenums{20}\%,\oldstylenums{40}\%,\oldstylenums{60}\%,\oldstylenums{80}\%,\oldstylenums{100}\%},
	ylabel={Accuracy},
	grid,
	legend columns=2,
	  legend style={
	    draw=none,
	    column sep=1ex,
	  },
	font=\scriptsize,
	scaled x ticks=false,
	yminorticks=false,
	axis x line=bottom,axis y line=left,
    grid style={line width=.1pt, draw=gray!10},
    major grid style={line width=.2pt,draw=gray!50},
    minor tick num=5,
    ymajorgrids,
	yminorgrids,
]
	\addplot[color=expred] table [x index = {0}, y index = {1}, col sep=comma, mark=none]
	{summaries/pp_#2.arff};
	\addplot[color=expblue] table [x index = {0}, y index = {2}, col sep=comma, mark=none]
	{summaries/pp_#2.arff};
\end{axis}
\end{tikzpicture}
}
